348 LONG_INT getMemory()
349 {
350 map<string, map<string, map<string, float> > >::iterator it1;
351 map<string, map<string, float> >::iterator it2;
352 map<string, float>::iterator it3;
353
354 LONG_INT memory = (LONG_INT)sizeof(siteTrans);
355
356 for (it1 = siteTrans.begin(); it1 != siteTrans.end(); ++it1)
357 {
358 memory += (LONG_INT)(sizeof((*it1).first) + (*it1).first.length()); // station
359 memory += (LONG_INT)sizeof((*it1).second);
360 for (it2 = (*it1).second.begin(); it2 != (*it1).second.end(); ++it2)
361 {
362 memory += (LONG_INT)(sizeof((*it2).first) + (*it2).first.length()); // channel
363 memory += (LONG_INT)sizeof((*it2).second);
364 for (it3 = (*it2).second.begin(); it3 != (*it2).second.end(); ++it3)
365 {
366 memory += (LONG_INT)(sizeof((*it3).first) + (*it3).first.length()); // band
367 memory += (LONG_INT)sizeof(float);
368 }
369 }
370 }
371 return memory + GeoTessModel::getMemory();
372 }

433 double getPathQ(GeoTessGreatCircle& path, const string& band,
434 const GeoTessInterpolatorType& interpolatorType = GeoTessInterpolatorType::LINEAR)
435 {
436 int level = getGrid().getNLevels(0)-1;
437 // convert level index into grid spacing in degrees.
438 double gridSpacing = exp(log(64.0)-level*log(2.0));
439 // set the integration interval to 1/10th the grid spacing, in radians
440 double integrationInterval = CPPUtils::toRadians(gridSpacing/10.);
441
442 GeoTessPointMap* pm = getPointMap();
443 int pointIndex;
444 double weight;
445 double integral = 0, pathLength=0.;
446
447 map<int, double> weights;
448 getWeights(path, integrationInterval, -1., interpolatorType, weights);
449
450
451 string attribute = "Q["+band+"]";
452 int bandIndex = getMetaData().getAttributeIndex(attribute);
453 if (bandIndex >= 0)
454 {
455 for (map<int, double>::iterator it = weights.begin(); it != weights.end(); it++)
456 {
457 pointIndex = (*it).first;
458 if (pointIndex < 0) return NaN_DOUBLE;
459
460 weight = (*it).second;
461
462 integral += weight/pm->getPointValueDouble(pointIndex, bandIndex);
463 pathLength += weight;
464 }
465 return pathLength/integral;
466 }
467 else
468 {
469 int q0Index = getMetaData().getAttributeIndex("Q0");
470 int etaIndex = getMetaData().getAttributeIndex("ETA");
471 if (q0Index < 0 || etaIndex < 0)
472 {
473 ostringstream os;
474 os << endl << "ERROR in GeoTessModelAmplitude::getPathIntegral" << endl
475 << "Model does not support attributes Q0/ETA or " << attribute
476 << ". Supported attributes include:" << endl;
477 for (int i=0; i<getNAttributes(); ++i)
478 os << " " << i << ": " << getMetaData().getAttributeName(i) << endl;
479
480 throw GeoTessException(os, __FILE__, __LINE__, 6005);
481 }
482
483 map<string, vector<float> >::iterator freq = frequencyMap.find(band);
484
485 if (freq == frequencyMap.end())
486 {
487 ostringstream os;
488 os << endl << "ERROR in GeoTessModelAmplitude::getPathIntegral" << endl
489 << band << "is not a supported frequency band. Supported attributes include:" << endl;
490 for (int i=0; i<getNAttributes(); ++i)
491 os << " " << i << ": " << getMetaData().getAttributeName(i) << endl;
492
493 throw GeoTessException(os, __FILE__, __LINE__, 6005);
494 }
495
496 double centerFrequency = sqrt((*freq).second[0] * (*freq).second[1]);
497
498 if (centerFrequency < 1e-6)
499 {
500 ostringstream os;
501 os << endl << "ERROR in GeoTessModelAmplitude::getPathIntegral" << endl
502 << "centerFrequency = " << centerFrequency << " is too small." << endl;
503 throw GeoTessException(os, __FILE__, __LINE__, 6005);
504 }
505
506 double q0, eta;
507
508 for (map<int, double>::iterator it = weights.begin(); it != weights.end(); it++)
509 {
510 pointIndex = (*it).first;
511 if (pointIndex < 0) return NaN_DOUBLE;
512
513 weight = (*it).second;
514
515 q0 = pm->getPointValueDouble(pointIndex, q0Index);
516 eta = pm->getPointValueDouble(pointIndex, etaIndex);
517
518 integral += weight/(q0 * pow(centerFrequency, eta));
519 pathLength += weight;
520
521 }
522 return pathLength/integral;
523 }
524 }

560 float getSiteTrans(const string& station, const string& channel, const string& band)
561 {
562 map<string, map<string, map<string, float> > >::iterator it1 = siteTrans.find(station);
563 if (it1 != siteTrans.end())
564 {
565 map<string, map<string, float> >::iterator it2 = (*it1).second.find(channel);
566 if (it2 != (*it1).second.end())
567 {
568 map<string, float>::iterator it3 = (*it2).second.find(band);
569 if (it3 != (*it2).second.end())
570 return (*it3).second;
571 }
572 }
573 return NaN_FLOAT;
574 }
