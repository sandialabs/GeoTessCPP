244 double getSpreadingESHmod(const double& distance)
245 {
246 // in case xt not set, not in Street-Herrmann
247 if (std::isnan(xt))
248 xt = 1;
249
250 double xstart = xc/xt;
251 double xend = xc*xt;
252
253 // watch out for tiny distances
254 double x = max(distance, 0.001);
255
256 if (x <= xstart) {
257 return -1*p1*log10(x);
258 } else if (x >= xend) {
259 double dp = p2 - p1;
260 return -1*p1*log10(xstart) - (p1 + dp/2)*log10(xend/xstart) -
261 p2*log10(x/xend);
262 } else {
263 // singular if xt=1, but should not get here
264 double p = (p2 - p1)/log10(xend/xstart);
265 double dp = p*log10(x/xstart);
266 return -1*p1*log10(xstart) - (p1 + dp/2)*log10(x/xstart);
267 }

278 double getSpreadingXY(const double& frequency, const double& distance)
279 {
280 if (std::isnan(frequency) || frequency <= 0)
281 {
282 ostringstream os;
283 os << endl << "ERROR in GeoTessModelAmplitude::getSpreadingXY()"<< endl
284 << "frequency = " << frequency << endl;
285 throw GeoTessException(os, __FILE__, __LINE__, 12001);
286 }
287
288 double logf = log10(frequency);
289 double logf2 = logf*logf;
290 double logx = log10(distance);
291 double logx2 = logx*logx;
292
293 double* c = getPhase() == "Pn" ? yang2007[0]
294 : getPhase() == "Sn" ? yang2007[1] : NULL;
295
296 if (!c)
297 {
298 ostringstream os;
299 os << endl << "ERROR in GeoTessModelAmplitude::getSpreadingXY()"<< endl
300 << "Phase = " << phase << " is invalid. Only Pn and Sn are recognized in this context." << endl;
301 throw GeoTessException(os, __FILE__, __LINE__, 12001);
302 }
303
304 return (c[0]*logf2 + c[1]*logf + c[2])*logx2 - (c[3]*logf2 + c[4]*logf +
305 c[5])*logx + (c[6]*logf2 + c[7]*logf + c[8]);
306 }

318 {
319 if (std::isnan(frequency) || frequency <= 0)

323 << "frequency = " << frequency << endl;

326
327 double logf = log10(frequency);
328 double mlogd = log10(1/distance);
329
330 double* params = yang2011[0];
331
332 double x1 = params[0]*logf + params[1];
333 double x2 = params[2]*logf + params[3];
334 double x3 = params[4]*logf + params[5];
335 return x1*mlogd*mlogd + x2*mlogd + x3;

582 virtual LONG_INT getMemory()
583 {
584 LONG_INT memory = GeoTessModel::getMemory();
585
586 memory += sizeof(phase) + phase.length();
587
588 memory += sizeof(frequencyMap);
589 map<string, vector<float> >::iterator it;
590 for (it = frequencyMap.begin(); it != frequencyMap.end(); ++it)
591 {
592 string f = (*it).first;
593 memory += (LONG_INT)(sizeof(f) + f.length());
594 vector<float> v = (*it).second;
595 memory += sizeof(v) + v.size()*sizeof(float);
596 }
597
598 memory += (LONG_INT)sizeof(siteTrans);
599 map<string, map<string, map<string, float> > >::iterator it1;
600 map<string, map<string, float> >::iterator it2;
601 map<string, float>::iterator it3;
602 for (it1 = siteTrans.begin(); it1 != siteTrans.end(); ++it1)
603 {
604 memory += (LONG_INT)(sizeof((*it1).first) + (*it1).first.length()); // station
605 memory += (LONG_INT)sizeof((*it1).second);
606 for (it2 = (*it1).second.begin(); it2 != (*it1).second.end(); ++it2)
607 {
608 memory += (LONG_INT)(sizeof((*it2).first) + (*it2).first.length()); // channel
609 memory += (LONG_INT)sizeof((*it2).second);
610 for (it3 = (*it2).second.begin(); it3 != (*it2).second.end(); ++it3)
611 {
612 memory += (LONG_INT)(sizeof((*it3).first) + (*it3).first.length()); // band
613 memory += (LONG_INT)sizeof(float);
614 }
615 }
616 }
617
618 memory += sizeof(yang2007)+18*sizeof(double);
619 memory += sizeof(yang2011)+18*sizeof(double);
620
621 memory += sizeof(interpolatorType) + interpolatorType.length();
622 memory += sizeof(dxkm);
623 memory += sizeof(spreadmode) + spreadmode.length();
624 memory += sizeof(rhos);
625 memory += sizeof(alphas);
626 memory += sizeof(betas);
627 memory += sizeof(radpatp);
628 memory += sizeof(radpats);
629 memory += sizeof(m0ref);
630 memory += sizeof(zeta);
631 memory += sizeof(sigma);
632 memory += sizeof(psi);
633 memory += sizeof(xc);
634 memory += sizeof(xt);
635 memory += sizeof(p1);
636 memory += sizeof(p2);
637 memory += sizeof(kfact);
638
639 return memory;

696 double getPathQ(GeoTessGreatCircle& path, const string& band)
697 {
698
699// int level = getGrid().getNLevels(0)-1;
700// // convert level index into grid spacing in degrees.
701// double gridSpacing = exp(log(64.0)-level*log(2.0));
702// // set the integration interval to 1/10th the grid spacing, in radians
703// double integrationInterval = CPPUtils::toRadians(gridSpacing/10.);
704
705 double integrationInterval = dxkm / 6371.; // convert km to radians
706
707 GeoTessPointMap* pm = getPointMap();
708 int pointIndex;
709 double weight;
710 double integral = 0, pathLength=0.;
711
712 map<int, double> weights;
713 if (interpolatorType == "NATURAL_NEIGHBOR")
714 getWeights(path, integrationInterval, -1., GeoTessInterpolatorType::NATURAL_NEIGHBOR, weights);
715 else if (interpolatorType == "LINEAR")
716 getWeights(path, integrationInterval, -1., GeoTessInterpolatorType::LINEAR, weights);
717
718 string attribute = "Q["+band+"]";
719 int bandIndex = getMetaData().getAttributeIndex(attribute);
720 if (bandIndex >= 0)
721 {
722 for (map<int, double>::iterator it = weights.begin(); it != weights.end(); it++)
723 {
724 pointIndex = (*it).first;
725 if (pointIndex < 0) return NaN_DOUBLE;
726
727 weight = (*it).second;
728
729 integral += weight/pm->getPointValueDouble(pointIndex, bandIndex);
730 pathLength += weight;
731 }
732 return pathLength/integral;
733 }
734 else
735 {
736 int q0Index = getMetaData().getAttributeIndex("Q0");
737 int etaIndex = getMetaData().getAttributeIndex("ETA");
738 if (q0Index < 0 || etaIndex < 0)
739 {
740 ostringstream os;
741 os << endl << "ERROR in GeoTessModelAmplitude::getPathQ" << endl
742 << "Model does not support attributes Q0/ETA or " << attribute
743 << ". Supported attributes include:" << endl;
744 for (int i=0; i<getNAttributes(); ++i)
745 os << " " << i << ": " << getMetaData().getAttributeName(i) << endl;
746
747 throw GeoTessException(os, __FILE__, __LINE__, 6005);
748 }
749
750 map<string, vector<float> >::iterator freq = frequencyMap.find(band);
751
752 if (freq == frequencyMap.end())
753 {
754 ostringstream os;
755 os << endl << "ERROR in GeoTessModelAmplitude::getPathQ" << endl
756 << band << "is not a supported frequency band. Supported attributes include:" << endl;
757 for (int i=0; i<getNAttributes(); ++i)
758 os << " " << i << ": " << getMetaData().getAttributeName(i) << endl;
759
760 throw GeoTessException(os, __FILE__, __LINE__, 6005);
761 }
762
763 double centerFrequency = sqrt((*freq).second[0] * (*freq).second[1]);
764
765 if (centerFrequency < 1e-6)
766 {
767 ostringstream os;
768 os << endl << "ERROR in GeoTessModelAmplitude::getPathQ" << endl
769 << "centerFrequency = " << centerFrequency << " is too small." << endl;
770 throw GeoTessException(os, __FILE__, __LINE__, 6005);
771 }
772
773 double q0, eta;
774
775 for (map<int, double>::iterator it = weights.begin(); it != weights.end(); it++)
776 {
777 pointIndex = (*it).first;
778 if (pointIndex < 0) return NaN_DOUBLE;
779
780 weight = (*it).second;
781
782 q0 = pm->getPointValueDouble(pointIndex, q0Index);
783 eta = pm->getPointValueDouble(pointIndex, etaIndex);
784
785 integral += weight/(q0 * pow(centerFrequency, eta));
786 pathLength += weight;
787
788 }
789 return pathLength/integral;
790 }
791 }

1347 double getQTerm(const string& station, const string& channel, const string& band,
1348 const double& rcvLat, const double& rcvLon,
1349 const double& sourceLat, const double& sourceLon)
1350 {
1351 // convert lats and lons into unit vectors.
1352 double rcv[3], src[3];
1353 getEarthShape().getVectorDegrees(rcvLat, rcvLon, rcv);
1354 getEarthShape().getVectorDegrees(sourceLat, sourceLon, src);
1355
1356 double earthRadius = (GeoTessUtils::getEarthRadius(src)
1357 + GeoTessUtils::getEarthRadius(rcv))/2;
1358
1359 // find the epicentral distance from src to rcv in km.
1360 double distance = GeoTessUtils::angle(src, rcv) * earthRadius;
1361
1362 // find average frequency of the specified band in Hz
1363 vector<float>& f = frequencyMap[band];
1364 double frequency = sqrt(f[0]*f[1]);
1365
1366 // get path integral of Q from src to rcv.
1367 double qpath = getPathQ(src, rcv, band);
1368
1369 //TODO: Scott: check this equation. It does not look right to me.
1370 double qterm = exp(2*PI*frequency*distance)/(2*qpath*pathvel);
1371
1372 // get the site term, or zero if station-channel-band not supported
1373 double siteTerm = getSiteTrans(station, channel, band);
1374 if (std::isnan(siteTerm)) siteTerm = 0.;
1375
1376 // get spreading term
1377 double spreading = getSpreading(distance, frequency);
1378
1379 return qterm + siteTerm + spreading;
1380 }
