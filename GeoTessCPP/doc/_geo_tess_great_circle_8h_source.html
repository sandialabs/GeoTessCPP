679  bool getIntersection(GeoTessGreatCircle& other, const bool& inRange, double* intersection)
680  {
681  if (GeoTessUtils::crossNormal(normal, other.normal, intersection) == 0.)
682  {
683  intersection[0] = intersection[1] = intersection[2] = NaN_DOUBLE;
684  return false;
685  }
686 
687  if (GeoTessUtils::scalarTripleProduct(firstPoint, intersection, normal) < 0.)
688  {
689  intersection[0] = -intersection[0];
690  intersection[1] = -intersection[1];
691  intersection[2] = -intersection[2];
692  }
693 
694  if (inRange && (getDistance(intersection) >= getDistance()
695  || other.getDistance(intersection) >= other.getDistance()))
696  {
697  intersection[0] = intersection[1] = intersection[2] = NaN_DOUBLE;
698  return false;
699  }
700 
701  return true;
702  }
