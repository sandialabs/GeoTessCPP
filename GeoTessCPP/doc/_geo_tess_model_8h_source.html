270  void loadGrid(T& input, const string& inputDirectory,
271  const string& relGridFilePath, const string& gridFileName,
272  const string& gridID, const string& funcName)
273  {
274  // process grid
275 
276  grid = NULL;
277  map<string, GeoTessGrid*>::iterator it = reuseGridMap.find(gridID);
278  if (it != reuseGridMap.end())
279  grid = it->second;
280 
281  if (gridFileName == "*")
282  {
283  // load the grid from this input file. The grid has to be read from the
284  // file, even a reference was retrieved from the reuseGridMap, so that the
285  // file is positioned where classes that extend GeoTessModel can read
286  // additional data.
287  GeoTessGrid* g = new GeoTessGrid(input);
288  if (!grid)
289  {
290  grid = g;
291 
292  grid->setGridInputFile(metaData->getInputModelFile());
293 
294  if (metaData->isGridReuseOn())
295  reuseGridMap[gridID] = grid;
296  }
297  else
298  delete g;
299  }
300  else if (!grid)
301  {
302  // build the name of the grid file using the input directory and
303  // the relative path to the grid file. Assume that both
304  // inputDirectory and relGridFilePath may be null or empty.
305  string gridFil = gridFileName;
306  if (relGridFilePath != "")
307  gridFil = CPPUtils::insertPathSeparator(relGridFilePath, gridFileName);
308  if (inputDirectory != "")
309  gridFil = CPPUtils::insertPathSeparator(inputDirectory, gridFil);
310 
311  grid = new GeoTessGrid();
312  grid->loadGrid(gridFil);
313  if (metaData->isGridReuseOn())
314  reuseGridMap[gridID] = grid;
315 
316  // throw an error if the grid ID's are not equal
317 
318  if (grid->getGridID() != gridID)
319  {
320  ostringstream os;
321  os << endl << "ERROR in GeoTessModel::" + funcName << endl
322  << "gridIDs in model file and existingGrid are not equal: "
323  << endl << " Model File gridID = " << gridID << endl
324  << " Grid File gridID = " << grid->getGridID() << endl;
325  throw GeoTessException(os, __FILE__, __LINE__, 1002);
326  }
327  }
328 
329  // add a reference to the grid and build the point map
330 
331  grid->addReference();
332  }

1431  template<typename T>
1432  void initializeData(const vector<string>& attributeNames,
1433  const vector<string>& attributeUnits, T fillValue)
1434  {
1435  if (profiles == NULL)
1436  {
1437  ostringstream os;
1438  os << endl << "ERROR in GeoTessModel::initializeData" << endl
1439  << "Attempting to initialize the model data before Profiles" << endl
1440  << "have been specified (profiles == NULL)" << endl;
1441  throw GeoTessException(os, __FILE__, __LINE__, 1001);
1442  }
1443 
1444  const GeoTessDataType& newDataType = GeoTessDataType::getDataType(fillValue);
1445  int nAttributesNew = attributeNames.size();
1446 
1447  int newDataTypeOrdinal = newDataType.ordinal();
1448  int oldDataTypeOrdinal = metaData->getDataType().ordinal();
1449 
1450  // if neither the dataType or the number of attributes has changed, there is no need to
1451  // copy the data. The only thing that happens is that the attributeNames or attributeUnits might change.
1452  if (newDataTypeOrdinal != oldDataTypeOrdinal || nAttributesNew != metaData->getNAttributes())
1453  {
1454  // initialize an array of the new dataType with number of elements
1455  // equal to the number of new attributes and populate it with fillValue.
1456  T* newValues = new T[nAttributesNew];
1457  for (int i=0; i<nAttributesNew; ++i)
1458  newValues[i] = fillValue;
1459 
1460  GeoTessProfile* profile;
1461  vector<GeoTessData*> data;
1462  for (int v = 0; v < getNVertices(); ++v)
1463  for (int lid = 0; lid < getNLayers(); ++lid)
1464  {
1465  profile = profiles[v][lid];
1466  data.reserve(profile->getNData());
1467  data.clear();
1468  for (int n = 0; n < profile->getNData(); ++n)
1469  {
1470  profile->getData(n)->getValues(newValues, nAttributesNew);
1471  data.push_back(GeoTessData::getData(newValues, nAttributesNew));
1472  }
1473  // set new entries into profile and continue
1474  profile->setData(data);
1475  }
1476  delete[] newValues;
1477  }
1478 
1479  metaData->setDataType(newDataType);
1480 
1481  metaData->setAttributes(attributeNames, attributeUnits);
1482  }
