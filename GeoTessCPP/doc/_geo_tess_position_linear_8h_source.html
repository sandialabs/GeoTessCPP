84  virtual void update2D(int tid)
85  {
86  vector<int>& vt = vertices[tid];
87  vector<double>& hc = hCoefficients[tid];
88 
89  vt.clear();
90  hc.clear();
91 
92  const int* trngl = grid.getTriangleVertexIndexes(getTriangle(tid));
93 
94  // if the interpolation point falls on a grid node set the list of interpolation
95  // vertices to include only the identified vertex, and the interpolation coefficient = 1.
96  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[0])) > cos(1e-7))
97  { vt.push_back(trngl[0]); hc.push_back(1.0); return; }
98 
99  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[1])) > cos(1e-7))
100  { vt.push_back(trngl[1]); hc.push_back(1.0); return; }
101 
102  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[2])) > cos(1e-7))
103  { vt.push_back(trngl[2]); hc.push_back(1.0); return; }
104 
105  // interpolation point is not coincident with a grid node.
106  vector<double>& lc = linearCoefficients[tid];
107  vt.push_back(trngl[0]);
108  hc.push_back(lc[0]);
109  vt.push_back(trngl[1]);
110  hc.push_back(lc[1]);
111  vt.push_back(trngl[2]);
112  hc.push_back(lc[2]);
113 }
