83  virtual void update2D(int tid)
84  {
85  vector<int>& vt = vertices[tid];
86  vector<double>& hc = hCoefficients[tid];
87 
88  vt.clear();
89  hc.clear();
90 
91  const int* trngl = grid.getTriangleVertexIndexes(getTriangle(tid));
92 
93  // if the interpolation point falls on a grid node set the list of interpolation
94  // vertices to include only the identified vertex, and the interpolation coefficient = 1.
95  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[0])) > cos(1e-7))
96  { vt.push_back(trngl[0]); hc.push_back(1.0); return; }
97 
98  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[1])) > cos(1e-7))
99  { vt.push_back(trngl[1]); hc.push_back(1.0); return; }
100 
101  if (GeoTessUtils::dot(unitVector, grid.getVertex(trngl[2])) > cos(1e-7))
102  { vt.push_back(trngl[2]); hc.push_back(1.0); return; }
103 
104  // interpolation point is not coincident with a grid node.
105  vector<double>& lc = linearCoefficients[tid];
106  vt.push_back(trngl[0]);
107  hc.push_back(lc[0]);
108  vt.push_back(trngl[1]);
109  hc.push_back(lc[1]);
110  vt.push_back(trngl[2]);
111  hc.push_back(lc[2]);
112 }
