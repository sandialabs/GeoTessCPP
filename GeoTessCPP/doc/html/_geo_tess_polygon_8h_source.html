377 bool contains(const double* x)
378 {
379 if (global || GeoTessUtils::dot(referencePoint, x) > cos(TOLERANCE))
380 return referenceIn;
381
382 GeoTessGreatCircle gcRef(referencePoint, x);
383
384 return onBoundary(gcRef) || ((edgeCrossings(gcRef) % 2 == 0) == referenceIn);
385 }

394 bool onBoundary(GeoTessGreatCircle& gcRef)
395 {
396 // point being evaluated is gcRef.getLast()
397 for (int i=0; i<(int)edges.size(); ++i)
398 {
399 // if point is very close to a polygon point, return true
400 if (GeoTessUtils::dot(gcRef.getLast(), edges[i]->getFirst()) >= cos(TOLERANCE))
401 return true;
402
403 // if gcRef and edge are coincident, i.e., their normals are parallel or
404 // anti-parallel, and the point being evaluated is between first and last
405 // point of the edge, then return true;
406 if (abs(GeoTessUtils::dot(gcRef.getNormal(), edges[i]->getNormal())) >= cos(TOLERANCE)
407 && edges[i]->getDistance(gcRef.getLast()) <= edges[i]->getDistance())
408 return true;
409 }
410 return false;
411 }

478 void getPoints(vector<double*> &points, const bool &repeatFirstPoint, const double &maxSpacing)
479 {
480 points.reserve(points.size()+edges.size()+1);
481
482 int n;
483 double dx;
484 for (int i = 0; i < (int)edges.size(); ++i)
485 {
486 n = (int)ceil(edges[i]->getDistance()/maxSpacing);
487 dx = edges[i]->getDistance()/n;
488 for (int j=0; j<n; ++j)
489 points.push_back(edges[i]->getPoint(j*dx));
490 }
491
492 if (repeatFirstPoint)
493 {
494 double* point = new double[3];
495 point[0] = edges[0]->getFirst()[0];
496 point[1] = edges[0]->getFirst()[1];
497 point[2] = edges[0]->getFirst()[2];
498 points.push_back(point);
499 }

526 double getArea()
527 {
528 double a, area = 0;
529 GeoTessGreatCircle* edge;
530 GeoTessGreatCircle* previous = edges[edges.size()-1];
531
532 for (int i=0; i<(int)edges.size(); ++i)
533 {
534 edge = edges[i];
535 a = PI - GeoTessUtils::angle(previous->getNormal(), edge->getNormal());
536
537 if (GeoTessUtils::scalarTripleProduct(previous->getNormal(), edge->getNormal(),
538 edge->getFirst()) < 0)
539 area += a;
540 else
541 area += 2*PI - a;
542
543 previous = edge;
544 }
545 area -= (edges.size()-2)*PI;
546 return area;
547 }
