525inline bool IFStreamAscii::read(string& token)
526{
527 string ln;
528
529 // if the strTokenPtr is >= than strTokens.size() then get more tokens
530
531 if (strTokenPtr >= (int) strTokens.size())
532 {
533 // get more tokens ... reset token pointer and list
534
535 strTokenPtr = 0;
536 strTokens.clear();
537
538 // read another data line from the stream ... if eof() is reached return
539 // true
540
541 if (!readLine(ln)) return false;
542
543 // tokennize the line into the tokens list
544
545 tokenize(ln, strTokens);
546 }
547
548 // get the token ... increment the pointer ... and return false for success
549
550 token = strTokens[strTokenPtr++];
551 return true;
552}

602inline bool IFStreamAscii::readByte(byte& b)
603{
604 string sb;
605
606 // read token into sb ... if eof or stream not open return false
607
608 if (!read(sb)) return false;
609
610 // attempt to scan token into b ... if unable issue error and return false
611
612 if (sscanf(sb.c_str(), "%hhd", &b) != 1)
613 {
614 ostringstream os;
615 os << endl << "ERROR in IFStreamAscii::readByte" << endl
616 << " Could Not Scan Byte From Token = " << sb << endl
617 << " On File Line: " << strTotlLinesRead << " ..." << endl;
618 throw GeoTessException(os, __FILE__, __LINE__, 9201);
619 }
620
621 // successful ... return true
622
623 return true;
624}
