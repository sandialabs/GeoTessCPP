365 LONG_INT getMemory()
366 {
367 LONG_INT memory = (LONG_INT) sizeof(GeoTessMetaData);
368
369 // EarthShape earthShape;
370 memory += (LONG_INT) sizeof(EarthShape);
371
372 memory += (LONG_INT) (description.length() + attributeFilterString.length()
373 + inputModelFile.length() + inputGridFile.length() + outputModelFile.length()
374 + outputGridFile.length() + modelSoftwareVersion.length() + modelGenerationDate.length());
375
376 // string* layerNames;
377 memory += nLayers * (LONG_INT) sizeof(string);
378 for (int i=0; i<nLayers; ++i)
379 memory += (LONG_INT) layerNames[i].length();
380
381 // int* layerTessIds;
382 memory += nLayers * (LONG_INT)sizeof(int);
383
384 // GeoTessDataType* dataType;
385 memory += nLayers * (int)sizeof(GeoTessDataType);
386
387 // string* attributeNames;
388 // string* attributeUnits;
389 if (nAttributes > 0)
390 {
391 memory += 2 * nAttributes * (LONG_INT) sizeof(string);
392 for (int i=0; i<nAttributes; ++i)
393 {
394 memory += (LONG_INT) attributeNames[i].length();
395 memory += (LONG_INT) attributeUnits[i].length();
396 }
397 }
398
399 // vector<int> attributeFilter;
400 memory += (LONG_INT) (attributeFilter.capacity() * sizeof(int));
401
402 // vector<int> inputFilter;
403 memory += (LONG_INT) (inputFilter.capacity() * sizeof(int));
404
405 return memory;
406 }

858 void setAttributes(const string& nms, const string& unts)
859 {
860 vector<string> names, units;
861 CPPUtils::split(nms, ";", names);
862 CPPUtils::split(unts, ";", units);
863 if (names.size() != units.size())
864 {
865 ostringstream os;
866 os << "Error in GeoTessMetaData::setAttributes(const string& nms, const string& unts)" << endl
867 << "nms = '" << nms << "'" << endl
868 << "unts = '" << unts << "'" << endl
869 << "Attribute names size (" << names.size()
870 << ") is not equal to units size (" << units.size() << ")" << endl;
871 names.clear(); units.clear();
872 throw GeoTessException(os, __FILE__, __LINE__, 6009);
873 }
874 setAttributes(names, units);
875
876 properties["attributeNames"] = nms;
877 properties["attributeUnits"] = unts;
878 }

1167 void setEulerRotationAngles(double* angles)
1168 {
1169 if (eulerRotationAngles != NULL)
1170 {
1171 delete [] eulerRotationAngles;
1172 eulerRotationAngles = NULL;
1173 CPPUtils::delete2DArray(eulerGridToModel);
1174 CPPUtils::delete2DArray(eulerModelToGrid);
1175 }
1176 if (angles != NULL)
1177 {
1178 eulerRotationAngles = angles;
1179
1180 // convert from degrees to radians
1181 eulerRotationAngles[0] *= DEG_TO_RAD;
1182 eulerRotationAngles[1] *= DEG_TO_RAD;
1183 eulerRotationAngles[2] *= DEG_TO_RAD;
1184
1185 eulerGridToModel = GeoTessUtils::getEulerMatrix(eulerRotationAngles);
1186 eulerModelToGrid = GeoTessUtils::inverse_3x3(eulerGridToModel);
1187
1188 // convert from radians back to degrees
1189 eulerRotationAngles[0] *= RAD_TO_DEG;
1190 eulerRotationAngles[1] *= RAD_TO_DEG;
1191 eulerRotationAngles[2] *= RAD_TO_DEG;
1192 }
1193 }
