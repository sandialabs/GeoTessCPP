500 void getPointVector(int pointIndex, double* v)
501 {
502 vector<int>& map = pointMap[pointIndex];
503 const double* vv = grid.getVertex(map[0]);
504 double r = profiles[map[0]][map[1]]->getRadius(map[2]);
505 v[0] = vv[0] * r;
506 v[1] = vv[1] * r;
507 v[2] = vv[2] * r;
508 }

559 profiles[m1[0]][m1[1]]->getRadius(m1[2]), grid.getVertex(m2[0]),
560 profiles[m2[0]][m2[1]]->getRadius(m2[2]));
561 }
