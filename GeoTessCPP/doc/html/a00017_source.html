458inline void CPPUtils::removeEOL(string& s)

475inline string CPPUtils::insertPathSeparator(const string& dir, const string& name)
476{
477
478 string path = dir;
479
480 while (path.size() && (path[path.length() - 1] == CPPUtils::FILE_SEP))
481 path.erase(path.length() - 1, 1);
482
483 if (path.length() > 0)
484 path = path+CPPUtils::FILE_SEP;
485
486 string nm = name;
487 while (nm.size() && (nm[0] == CPPUtils::FILE_SEP))
488 nm.erase((unsigned)0, 1);
489
490 return path+nm;
491}

496template <typename T>
497void CPPUtils::minmax(const vector<T>& v, T& mn, T& mx)
498{
499 // exit without modification if v is empty
500
501 if (v.size())
502 {
503 // assign mx and mn to first entry
504
505 mx = mn = v[0];
506
507 // adjust mx and mn based on remaining entries
508
509 for (int i = 1; i < v.size(); i++)
510 {
511 if (mn > v[i]) mn = v[i];
512 if (mx < v[i]) mx = v[i];
513 }
514 }
515}
