280 template<class T>
281 void loadGrid(T& input, const string& inputDirectory,
282 const string& relGridFilePath, const string& gridFileName,
283 const string& gridID, const string& funcName)
284 {
285 // process grid
286
287 grid = NULL;
288 map<string, GeoTessGrid*>::iterator it = reuseGridMap.find(gridID);
289 if (it != reuseGridMap.end())
290 grid = it->second;
291
292 if (gridFileName == "*")
293 {
294 // load the grid from this input file. The grid has to be read from the
295 // file, even a reference was retrieved from the reuseGridMap, so that the
296 // file is positioned where classes that extend GeoTessModel can read
297 // additional data.
298 GeoTessGrid* g = new GeoTessGrid(input);
299 if (!grid)
300 {
301 grid = g;
302
303 grid->setGridInputFile(metaData->getInputModelFile());
304
305 if (metaData->isGridReuseOn())
306 reuseGridMap[gridID] = grid;
307 }
308 else
309 delete g;
310 }
311 else if (!grid)
312 {
313 // build the name of the grid file using the input directory and
314 // the relative path to the grid file. Assume that both
315 // inputDirectory and relGridFilePath may be null or empty.
316 string gridFil = gridFileName;
317 if (relGridFilePath != "")
318 gridFil = CPPUtils::insertPathSeparator(relGridFilePath, gridFileName);
319 if (inputDirectory != "")
320 gridFil = CPPUtils::insertPathSeparator(inputDirectory, gridFil);
321
322 grid = new GeoTessGrid();
323 grid->loadGrid(gridFil);
324 if (metaData->isGridReuseOn())
325 reuseGridMap[gridID] = grid;
326
327 // throw an error if the grid ID's are not equal
328
329 if (grid->getGridID() != gridID)
330 {
331 ostringstream os;
332 os << endl << "ERROR in GeoTessModel::" + funcName << endl
333 << "gridIDs in model file and existingGrid are not equal: "
334 << endl << " Model File gridID = " << gridID << endl
335 << " Grid File gridID = " << grid->getGridID() << endl;
336 throw GeoTessException(os, __FILE__, __LINE__, 1002);
337 }
338 }
339
340 // add a reference to the grid and build the point map
341
342 grid->addReference();
343 }

591 static LONG_INT getReuseGridMapMemory()
592 {
593 // this ignores some memory allocated to support the map. It is complicated and
594 // platform dependent. Also probably small compared to the size of the Grids.
595 LONG_INT memory = sizeof(map<string, GeoTessGrid*>);
596
597 memory += (LONG_INT) (reuseGridMap.size() * (sizeof(string) + sizeof(GeoTessGrid*)));
598
599 for (map<string, GeoTessGrid*>::iterator it = reuseGridMap.begin(); it != reuseGridMap.end(); it++)
600 memory += (LONG_INT)it->first.length() + it->second->getMemory();
601
602 return memory;
603 }

1404 void profileCount(vector< vector<int> >& count)
1405 {
1406 count.clear();
1407
1408 getPointMap();
1409
1410 // this is total number of profiles of each type, independent of layer index.
1411 vector<int> totalCount;
1412 for (int profileType=0; profileType<GeoTessProfileType::size(); ++profileType)
1413 totalCount.push_back(0);
1414
1415 for (int layer = 0; layer < getNLayers(); ++layer)
1416 {
1417 vector<int> typeCount;
1418
1419 for (int profileType=0; profileType<GeoTessProfileType::size(); ++profileType)
1420 typeCount.push_back(0);
1421
1422 count.push_back(typeCount);
1423 }
1424
1425 for (int layer = 0; layer < getNLayers(); ++layer)
1426 {
1427 for (int vertex = 0; vertex < metaData->getNVertices(); ++vertex)
1428 {
1429 int pType = profiles[vertex][layer]->getType().ordinal();
1430 count[layer][pType] += 1;
1431 totalCount[pType] += 1;
1432 }
1433 }
1434
1435 count.push_back(totalCount);
1436 }

1471 template<typename T>
1472 void initializeData(const vector<string>& attributeNames,
1473 const vector<string>& attributeUnits, T fillValue)
1474 {
1475 if (profiles == NULL)
1476 {
1477 ostringstream os;
1478 os << endl << "ERROR in GeoTessModel::initializeData" << endl
1479 << "Attempting to initialize the model data before Profiles" << endl
1480 << "have been specified (profiles == NULL)" << endl;
1481 throw GeoTessException(os, __FILE__, __LINE__, 1001);
1482 }
1483
1484 const GeoTessDataType& newDataType = GeoTessDataType::getDataType(fillValue);
1485 int nAttributesNew = attributeNames.size();
1486
1487 int newDataTypeOrdinal = newDataType.ordinal();
1488 int oldDataTypeOrdinal = metaData->getDataType().ordinal();
1489
1490 // if neither the dataType or the number of attributes has changed, there is no need to
1491 // copy the data. The only thing that happens is that the attributeNames or attributeUnits might change.
1492 if (newDataTypeOrdinal != oldDataTypeOrdinal || nAttributesNew != metaData->getNAttributes())
1493 {
1494 // initialize an array of the new dataType with number of elements
1495 // equal to the number of new attributes and populate it with fillValue.
1496 T* newValues = new T[nAttributesNew];
1497 for (int i=0; i<nAttributesNew; ++i)
1498 newValues[i] = fillValue;
1499
1500 GeoTessProfile* profile;
1501 vector<GeoTessData*> data;
1502 for (int v = 0; v < getNVertices(); ++v)
1503 for (int lid = 0; lid < getNLayers(); ++lid)
1504 {
1505 profile = profiles[v][lid];
1506 data.reserve(profile->getNData());
1507 data.clear();
1508 for (int n = 0; n < profile->getNData(); ++n)
1509 {
1510 profile->getData(n)->getValues(newValues, nAttributesNew);
1511 data.push_back(GeoTessData::getData(newValues, nAttributesNew));
1512 }
1513 // set new entries into profile and continue
1514 profile->setData(data);
1515 }
1516 delete[] newValues;
1517 }
1518
1519 metaData->setDataType(newDataType);
1520
1521 metaData->setAttributes(attributeNames, attributeUnits);
1522 }
