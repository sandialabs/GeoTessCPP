212 void updateRadialCoefficients()
213 {
214 // make sure dimensions of radialIndexes and radialCoefficients
215 // are at least as big as the number of vertices involved in interpolation.
216 if (radialIndexes.size() < vertices[tessid].size())
217 {
218 radialIndexes.resize(vertices[tessid].size());
219 radialCoefficients.resize(vertices[tessid].size());
220 }
221
222 // if radial interpolation coefficients have already been computed
223 // do nothing.
224 if (radialIndexes[0].size() == 0)
225 {
226 vector<int>& v = vertices[tessid];
227 for (int i = 0; i < (int)v.size(); ++i)
228 modlProfiles[v[i]][layerId]->setInterpolationCoefficients(radialInterpolatorType,
229 radialIndexes[i], radialCoefficients[i], radius, radiusOutOfRangeAllowed);
230 }

259 int biggerTriangle(int t1, int t2)
260 {
261 const int* tv = grid.getTriangleVertexIndexes(t1);
262 double dot1 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
263 + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
264 + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
265
266 tv = grid.getTriangleVertexIndexes(t2);
267 double dot2 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
268 + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
269 + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
270
271 return dot2 > dot1 ? t1 : t2;
272

442 long memory()
443 {
444 long m = 0;
445
446 // memory for int* triangle, tessLevels and maxTessLevel.
447 m += (long)(grid.getNTessellations() * 3 * sizeof(int));
448
449 // vector<double> layerRadii;
450 m += (long)(layerRadii.capacity() * sizeof(double));
451
452 // vector< vector< int> > vertices;
453 m += (long)(vertices.capacity()*sizeof(vector<int>));
454 for (int i=0; i<(int)vertices.size(); ++i)
455 m += (long)(vertices[i].capacity()*sizeof(int));
456
457 // vector< vector< double> > linearCoefficients;
458 m += (long)(linearCoefficients.capacity()*sizeof(vector<double>));
459 for (int i=0; i<(int)linearCoefficients.size(); ++i)
460 m += (long)(linearCoefficients[i].capacity()*sizeof(double));
461
462 // vector< vector< double> > hCoefficients;
463 m += (long)(hCoefficients.capacity()*sizeof(vector<double>));
464 for (int i=0; i<(int)hCoefficients.size(); ++i)
465 m += (long)(hCoefficients[i].capacity()*sizeof(double));
466
467 // vector<vector<int> > radialIndexes;
468 m += (long)(radialIndexes.capacity()*sizeof(vector<int>));
469 for (int i=0; i<(int)radialIndexes.size(); ++i)
470 m += (long)(radialIndexes[i].capacity()*sizeof(int));
471
472 // vector<vector<double> > radialCoefficients;
473 m += (long)(radialCoefficients.capacity()*sizeof(vector<double>));
474 for (int i=0; i<(int)radialCoefficients.size(); ++i)
475 m += (long)(radialCoefficients[i].capacity()*sizeof(double));
476
477 return m;
