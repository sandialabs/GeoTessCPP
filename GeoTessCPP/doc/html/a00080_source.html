211 void updateRadialCoefficients()
212 {
213 // make sure dimensions of radialIndexes and radialCoefficients
214 // are at least as big as the number of vertices involved in interpolation.
215 if (radialIndexes.size() < vertices[tessid].size())
216 {
217 radialIndexes.resize(vertices[tessid].size());
218 radialCoefficients.resize(vertices[tessid].size());
219 }
220
221 // if radial interpolation coefficients have already been computed
222 // do nothing.
223 if (radialIndexes[0].size() == 0)
224 {
225 vector<int>& v = vertices[tessid];
226 double r = radius;
227 double depth = depthSpecified ? getDepth() : NaN_DOUBLE;
228 double vertex[3];
229 for (int i = 0; i < (int)v.size(); ++i)
230 {
231 if (depthSpecified)
232 {
233 model->getVertex(v[i], vertex);
234 r = model->getEarthShape().getEarthRadius(vertex)-depth;
235 }
236 modlProfiles[v[i]][layerId]->setInterpolationCoefficients(radialInterpolatorType,
237 radialIndexes[i], radialCoefficients[i], r, radiusOutOfRangeAllowed);
238 }
239 }

268 int biggerTriangle(int t1, int t2)
269 {
270 const int* tv = grid.getTriangleVertexIndexes(t1);
271 double dot1 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
272 + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
273 + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
274
275 tv = grid.getTriangleVertexIndexes(t2);
276 double dot2 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
277 + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
278 + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
279
280 return dot2 > dot1 ? t1 : t2;

1316 void getBorehole(const double& rSpacing, bool computeDepth,
1317 vector<int>& layers, vector<double>& radii, vector<vector<double> >& attributes)
1318 {
1319 // save current layer and radius so they can be restored at the end.
1320 int layer0 = getLayerId();
1321 double r0 = getRadius();
1322
1323 layers.clear();
1324 radii.clear();
1325 attributes.clear();
1326
1327 // loop over the layers and compute radii such that there
1328 // are radii at the top and bottom of each layer.
1329 for (int layer=0; layer<model->getNLayers(); ++layer)
1330 {
1331 double rbot = getRadiusBottom(layer);
1332 double rtop = getRadiusTop(layer);
1333
1334 int nintervals = (int)ceil((rtop - rbot)/rSpacing);
1335 double dinterval = (rtop - rbot)/nintervals;
1336
1337 for (int i=0; i <= nintervals; ++i)
1338 {
1339 double r = rbot + i*dinterval;
1340 radii.push_back(r);
1341 layers.push_back(layer);
1342 }
1343 }
1344
1345 // loop over the radii and interpolate attribute values.
1346 for (int i=0; i<radii.size(); ++i)
1347 {
1348 setRadius(layers[i], radii[i]);
1349 vector<double> att(model->getNAttributes());
1350 for (int j=0; j<model->getNAttributes(); ++j)
1351 att[j] = getValue(j);
1352 attributes.push_back(att);
1353 }
1354
1355 // currently, monotonically increasing radii are stored.
1356 // if computeDepth is true, convert to monotonically increasing depths.
1357 if (computeDepth)
1358 {
1359 for (int i=0; i < radii.size()/2; ++i)
1360 {
1361 int j = radii.size()-1-i;
1362 int layer = layers[i];
1363 layers[i] = layers[j];
1364 layers[j] = layer;
1365
1366 double r = radii[i];
1367 radii[i] = radii[j];
1368 radii[j] = r;
1369
1370 for (int a=0; a < model->getNAttributes(); ++a)
1371 {
1372 double att = attributes[i][a];
1373 attributes[i][a] = attributes[j][a];
1374 attributes[j][a] = att;
1375 }
1376 }
1377 for (int i=0; i<radii.size(); ++i)
1378 radii[i] = getEarthRadius()-radii[i];
1379 }
1380
1381 // restore original layer and radius.
1382 setRadius(layer0, r0);
1383 }
