450 LONG_INT getMemory()
451 {
452 LONG_INT memory = (LONG_INT)sizeof(GeoTessGrid);
453
454 // double** vertices
455 memory += nVertices * (LONG_INT)sizeof(double*) + nVertices * 3 * (LONG_INT)sizeof(double);
456
457 // int** triangles
458 memory += nTriangles * (LONG_INT)sizeof(int*) + nTriangles * 3 * (LONG_INT)sizeof(int);
459
460 // int** levels
461 memory += nLevels * (LONG_INT)sizeof(int*) + nLevels * 2 * (LONG_INT)sizeof(int);
462
463 // int** tessellations
464 memory += nTessellations * (LONG_INT)sizeof(int*) + nTessellations * 2 * (LONG_INT)sizeof(int);
465
466 // int* descendants
467 memory += nTriangles * (LONG_INT)sizeof(int);
468
469 // vector<vector<Edge*> > edgeList : nTriangles x 3 array of Edges
470 memory += (LONG_INT) (edgeList.capacity() * sizeof(vector<Edge*>)
471 + edgeList.size() * 3 * (sizeof(Edge*) + sizeof(Edge)));
472
473 // vector<vector<Edge*> > spokeList;
474 memory += (LONG_INT) (spokeList.capacity() * sizeof(vector<Edge*>)
475 + spokeList.size() * 3 * sizeof(Edge*));
476
477 // vector<double*> circumCenters is nTriangles x 3 array of doubles
478 memory += (LONG_INT) (circumCenters.capacity() * sizeof(double*));
479 for (int i=0; i<(int)circumCenters.size(); ++i)
480 if (circumCenters[i]) memory += 3 * (LONG_INT)sizeof(double);
481
482 // An nLevels x nVertices x n array that stores the indices of the n
483 // triangles of which each vertex is a member.
484 // vector< vector< vector<int> > > vtxTriangles;
485 memory += (LONG_INT) (vtxTriangles.capacity() * sizeof(vector< vector<int> >));
486 for (int i=0; i<(int)vtxTriangles.size(); ++i)
487 {
488 memory += (LONG_INT) (vtxTriangles[i].capacity() * sizeof(vector<int>));
489 for (int j=0; j<(int)vtxTriangles[i].size(); ++j)
490 memory += (LONG_INT) (vtxTriangles[i][j].capacity() * sizeof(int));
491 }
492
493 // An nLevels x n array that stores the indexes of the n vertices
494 // that are connected together by triangles on the corresponding
495 // level. Lazy evaluation is used to construct this array of sets.
496 // vector< set<int> > connectedVertices;
497 memory += (LONG_INT) (connectedVertices.capacity() * sizeof(set<int>));
498 for (int i=0; i<(int)connectedVertices.size(); ++i)
499 memory += (LONG_INT) (connectedVertices[i].size() * sizeof(int));
500
501 // add memory requirements for all the string variables.
502 memory += (LONG_INT) (gridID.length() + gridInputFile.length() + gridOutputFile.length()
503 + gridSoftwareVersion.length() + gridGenerationDate.length());
504
505 return memory;
506 }

636 int findClosestVertex(double* unit_vector, int tessId)
637 {
638 int* t = triangles[getTriangle(getFirstTriangle(tessId, 0), unit_vector)];
639
640 int index = 0;
641 double dot = GeoTessUtils::dot(unit_vector, vertices[t[0]]);
642
643 double doti = GeoTessUtils::dot(unit_vector, vertices[t[1]]);
644 if (doti > dot)
645 {
646 index = 1;
647 dot = doti;
648 }
649
650 doti = GeoTessUtils::dot(unit_vector, vertices[t[2]]);
651 if (doti > dot)
652 {
653 index = 2;
654 dot = doti;
655 }
656
657 return t[index];
658 }
