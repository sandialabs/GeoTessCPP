655 static double* getVector(const double& lat, const double& lon, double* v)
656 {
657 // convert lat from geographic to geocentric latitude.
658 double temp = getGeocentricLat(lat);
659
660 // z component of v is sin of geocentric latitude.
661 v[2] = sin(temp);
662
663 // set lat = to cos of geocentric latitude
664 temp = cos(temp);
665
666 // compute x and y components of v
667 v[0] = temp * cos(lon);
668 v[1] = temp * sin(lon);
669
670 return v;
671 }

818 static bool vectorTripleProduct(const double* const v0,
819 const double* const v1, const double* const v2, double* const rslt)
820 {
821 // set q = v0 cross v1
822 double q0 = v0[1] * v1[2] - v0[2] * v1[1];
823 double q1 = v0[2] * v1[0] - v0[0] * v1[2];
824 double q2 = v0[0] * v1[1] - v0[1] * v1[0];
825
826 // set w = q cross v2
827 double w0 = q1 * v2[2] - q2 * v2[1];
828 double w1 = q2 * v2[0] - q0 * v2[2];
829 double w2 = q0 * v2[1] - q1 * v2[0];
830
831 // set rslt = w
832 rslt[0] = w0;
833 rslt[1] = w1;
834 rslt[2] = w2;
835
836 // normalize rslt to unit length. if the length
837 // of v1 or v2 is zero or they are nearly parallel then
838 // rslt will = {0,0,0} and the function will return false;
839 return normalize(rslt) != 0.;
840 }
