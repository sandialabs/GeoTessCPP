79typedef struct // GeoTessModelC
80{
81 // An array of pointers to c++ GeoTessModel objects and derived classes.
82 // If the geotess model being supported is an instance of a plain GeoTessModel
83 // base class then pointers[0] points to that object.
84 //
85 // If the geotess model being supported is an instance of class GeoTessModel_A that extends
86 // GeoTessModel, then pointer[0] is a GeoTessModel* and pointers[1] is a GeoTessModel_A*.
87 //
88 // If the geotess model being supported is an instance of a class GeoTessModel_B that extends
89 // GeoTessModel_A, then pointer[0] is a GeoTessModel*, pointers[1] is a GeoTessModel_A* and
90 // pointers[2] is a GeoTessModel_B.
91 //
92 // Up to 10 levels of inheritance are supported.
93 //
94 int size;
95 void* pointers[10];
