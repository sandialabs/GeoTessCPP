448 LibCorr3D* setInterpolationLocation(double lon, double lat, double depth = 0.0, double time =
449 NA_VALUE)
450 {
451 if (lat < -90 || lat > 90)
452 {
453 ostringstream os;
454 os << fixed << setprecision(6);
455 os << endl << "ERROR in LibCorr3D::setInterpolationLocation" << endl
456 << "latitude (" << lat << ") is out of range" << endl;
457 throw GeoTessException(os, __FILE__, __LINE__, 109);
458 }
459
460 GeoTessUtils::getVectorDegrees(lat, lon, lc3dCurrIntrpVector);
461 lc3dCurrIntrpRadius = GeoTessUtils::getEarthRadius(lc3dCurrIntrpVector)-depth;
462 lc3dCurrIntrpTime = time;
463 return this;
464 }
465
478 LibCorr3D* setInterpolationLocation(double* unitVector, double radius = -1., double time =
479 NA_VALUE)
480 {
481 lc3dCurrIntrpVector[0] = unitVector[0];
482 lc3dCurrIntrpVector[1] = unitVector[1];
483 lc3dCurrIntrpVector[2] = unitVector[2];
484 lc3dCurrIntrpRadius = radius < 0. ? GeoTessUtils::getEarthRadius(unitVector) : radius;
485 lc3dCurrIntrpTime = time;
486 return this;
487 }
