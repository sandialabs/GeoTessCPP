259 virtual LONG_INT getMemory()

263 memory += GeoTessModel::getMemory();
264
265 memory += (LONG_INT) (sta.length() + staRefsta.length() + phase.length() + parameters.length()

267 + supportedPhases.length() + staName.length() + staType.length());
